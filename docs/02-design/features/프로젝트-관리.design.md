### 1.3 간트 차트 구현 결정 (v2.0 업데이트)

**선택: 커스텀 SVG 구현 (frappe-gantt 미사용)**

> **설계 변경**: 초안에서는 `frappe-gantt` 기반 커스텀 확장을 계획했으나,
> 멀티 세그먼트 바, 베이스라인 오버레이, 마일스톤 다이아몬드 등 요구사항 충족을
> 위해 처음부터 완전한 커스텀 SVG 간트 차트로 구현.

| 기준 | 커스텀 SVG (최종 선택) | frappe-gantt (초안) | dhtmlx-gantt (상용) |
|------|----------------------|-------------------|-------------------|
| 라이선스 비용 | 무료 | 무료 | $749/개발자 |
| 다중 세그먼트 바 | 네이티브 지원 | 커스텀 렌더러 필요 | 미지원 |
| Baseline 오버레이 | SVG 레이어 완전 제어 | 제한적 | 플러그인 제공 |
| 마일스톤 다이아몬드 | 지원 | 제한적 | 지원 |
| 번들 크기 | 의존성 없음 | ~15KB | ~300KB |
| 커스터마이징 | 완전 제어 | 일부 제어 | 제한적 |

**구현된 커스텀 SVG 렌더링 기능** (`apps/web/src/app/projects/[id]/page.tsx`):
- `SegmentBar`: 한 행에 다수의 분리된 바 렌더링 (갭 점선 포함)
- `BaselineBarRenderer`: 반투명 회색 Baseline 바 SVG 레이어
- `DependencyArrow`: FS/SS/FF/SF 화살표 직접 그리기
- `CriticalPathHighlighter`: 크리티컬 패스 강조 (빨간 테두리)
- `MilestoneDiamond`: 마일스톤 다이아몬드 심볼
- `TodayLine`: 오늘 날짜 수직선

## 2. DB 스키마 설계 (Prisma Schema)

```prisma
// services/project/prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
  schemas  = ["project"]
}

// ─────────────────────────────────────────────
// 1. 프로젝트 기본 구조
// ─────────────────────────────────────────────

enum ProjectStatus {
  PLANNING
  IN_PROGRESS
  ON_HOLD
  COMPLETED
  CANCELLED

  @@schema("project")
}

enum TaskStatus {
  TODO
  IN_PROGRESS
  DONE
  BLOCKED

  @@schema("project")
}

enum DependencyType {
  FS  // Finish-to-Start
  SS  // Start-to-Start
  FF  // Finish-to-Finish
  SF  // Start-to-Finish

  @@schema("project")
}

enum AllocationMode {
  PERCENT
  HOURS

  @@schema("project")
}

enum ResourceType {
  PERSON
  EQUIPMENT

  @@schema("project")
}

enum GroupType {
  DEPARTMENT
  CLIENT
  CUSTOM

  @@schema("project")
}

enum ChangeType {
  SEGMENT_ADDED
  SEGMENT_REMOVED
  DATE_CHANGED
  ASSIGNMENT_CHANGED
  PROGRESS_UPDATED
  STATUS_CHANGED

  @@schema("project")
}

enum TemplateScope {
  ORGANIZATION
  PERSONAL

  @@schema("project")
}

model Project {
  id              String        @id @default(cuid())
  name            String
  description     String?
  status          ProjectStatus @default(PLANNING)
  plannedBudget   Decimal?      @db.Decimal(15, 2)
  actualBudget    Decimal?      @db.Decimal(15, 2)
  ownerId         String        // user-service userId
  createdBy       String
  createdAt       DateTime      @default(now())
  updatedAt       DateTime      @updatedAt

  milestones      Milestone[]
  tasks           Task[]
  baselines       ProjectBaseline[]
  activityLogs    ActivityLog[]
  groupMemberships ProjectGroupMembership[]

  @@map("projects")
  @@schema("project")
}

model Milestone {
  id          String   @id @default(cuid())
  projectId   String
  name        String
  description String?
  sortOrder   Int      @default(0)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  project     Project  @relation(fields: [projectId], references: [id], onDelete: Cascade)
  tasks       Task[]

  @@map("milestones")
  @@schema("project")
}

// Task = 논리적 컨테이너 (직접 날짜 없음)
model Task {
  id                  String      @id @default(cuid())
  projectId           String
  milestoneId         String?
  parentId            String?     // WBS 계층 구조 (상위 태스크 ID, null = 루트)
  name                String
  description         String?
  status              TaskStatus  @default(TODO)
  sortOrder           Int         @default(0)
  isMilestone         Boolean     @default(false) // 마일스톤 타입 태스크
  // effectiveStartDate = min(segments.startDate) — DB 저장 안 함, 계산값
  // effectiveEndDate   = max(segments.endDate)   — DB 저장 안 함, 계산값
  overallProgress     Float       @default(0)   // 0~100
  isManualProgress    Boolean     @default(false)
  isCritical          Boolean     @default(false) // CPM 결과
  totalFloat          Float?      // CPM Forward/Backward Pass 결과
  createdBy           String
  createdAt           DateTime    @default(now())
  updatedAt           DateTime    @updatedAt

  project             Project     @relation(fields: [projectId], references: [id], onDelete: Cascade)
  milestone           Milestone?  @relation(fields: [milestoneId], references: [id])
  parent              Task?       @relation("TaskHierarchy", fields: [parentId], references: [id])
  children            Task[]      @relation("TaskHierarchy")
  segments            TaskSegment[]
  predecessorDeps     TaskDependency[] @relation("predecessor")
  successorDeps       TaskDependency[] @relation("successor")
  comments            Comment[]
  attachments         Attachment[]
  scheduleHistories   TaskScheduleHistory[]
  taskBaselines       TaskBaseline[]

  @@map("tasks")
  @@schema("project")
}

// TaskSegment = 구간별 독립 일정 + 자원
model TaskSegment {
  id              String   @id @default(cuid())
  taskId          String
  name            String   // "지하 배관", "지상 배관", "수압 테스트"
  sortOrder       Int      @default(0)
  startDate       DateTime @db.Date
  endDate         DateTime @db.Date
  progressPercent Float    @default(0) // 0~100
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  task            Task     @relation(fields: [taskId], references: [id], onDelete: Cascade)
  assignments     SegmentAssignment[]
  baselineSegments TaskBaselineSegment[]

  // 같은 Task 내 세그먼트는 날짜 중첩 불가 — 애플리케이션 레벨 검증
  @@map("task_segments")
  @@schema("project")
}

// 세그먼트별 자원 배정 (% 또는 h/day 이중 모드)
model SegmentAssignment {
  id                      String         @id @default(cuid())
  segmentId               String
  resourceId              String         // resource.id
  allocationMode          AllocationMode @default(PERCENT)
  allocationPercent       Float?         // PERCENT 모드 값 (0~100)
  allocationHoursPerDay   Float?         // HOURS 모드 값
  createdAt               DateTime       @default(now())
  updatedAt               DateTime       @updatedAt

  segment                 TaskSegment    @relation(fields: [segmentId], references: [id], onDelete: Cascade)

  @@unique([segmentId, resourceId])
  @@map("segment_assignments")
  @@schema("project")
}

// 태스크 간 의존 관계 (CPM 입력)
model TaskDependency {
  id            String         @id @default(cuid())
  predecessorId String
  successorId   String
  type          DependencyType @default(FS)
  lagDays       Int            @default(0) // 음수 = 선행 가능

  predecessor   Task           @relation("predecessor", fields: [predecessorId], references: [id], onDelete: Cascade)
  successor     Task           @relation("successor", fields: [successorId], references: [id], onDelete: Cascade)

  @@unique([predecessorId, successorId])
  @@map("task_dependencies")
  @@schema("project")
}

// ─────────────────────────────────────────────
// 2. 자원 (Resource)
// ─────────────────────────────────────────────

model Resource {
  id                  String       @id @default(cuid())
  userId              String?      @unique // user-service userId (인력인 경우)
  name                String
  type                ResourceType @default(PERSON)
  dailyCapacityHours  Float        @default(8.0)
  isActive            Boolean      @default(true)
  createdAt           DateTime     @default(now())
  updatedAt           DateTime     @updatedAt

  groupMemberships    ResourceGroupMember[]

  @@map("resources")
  @@schema("project")
}

// 자원 그룹 계층 (팀/장비 분류)
model ResourceGroup {
  id          String   @id @default(cuid())
  name        String
  description String?
  parentId    String?  // 상위 그룹 (null = 루트)
  sortOrder   Int      @default(0)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  parent      ResourceGroup?  @relation("ResourceGroupHierarchy", fields: [parentId], references: [id])
  children    ResourceGroup[] @relation("ResourceGroupHierarchy")
  members     ResourceGroupMember[]

  @@map("resource_groups")
  @@schema("project")
}

// ResourceGroup ↔ Resource N:M 매핑
model ResourceGroupMember {
  id          String        @id @default(cuid())
  groupId     String
  resourceId  String
  createdAt   DateTime      @default(now())

  group       ResourceGroup @relation(fields: [groupId], references: [id], onDelete: Cascade)
  resource    Resource      @relation(fields: [resourceId], references: [id], onDelete: Cascade)

  @@unique([groupId, resourceId])
  @@map("resource_group_members")
  @@schema("project")
}

  @@schema("project")

// 태스크·세그먼트 일정 변경 이벤트 로그

  changedBy     String
  changedAt     DateTime   @default(now())
  changeReason  String     // 필수
  changeType    ChangeType
  field         String     // "startDate" | "endDate" | "allocationPercent" | "name" | ...

  newValue      String?

  @@schema("project")
```

### 4.1 데이터 구조

```typescript
// domain/algorithms/cpm.algorithm.ts

interface CpmNode {
  taskId: string;
  effectiveDuration: number;  // days (effectiveEnd - effectiveStart, Gap 포함)
  effectiveStart: Date;       // min(segments.startDate)
  effectiveEnd: Date;         // max(segments.endDate)

  // Forward Pass 결과
  earlyStart: Date;
  earlyFinish: Date;
```

```typescript
  // 2단계: Forward Pass
  forwardPass(nodes: CpmNode[], edges: CpmEdge[]): void {
    const sorted = topologicalSort(nodes, edges);
    for (const node of sorted) {
      const predecessors = getPredecessors(node.taskId, edges);
      if (predecessors.length === 0) {
        node.earlyStart = projectStartDate;
      } else {
        node.earlyStart = max(predecessors.map(p => {
          return calculateSuccessorConstraint(p, node, edges);
        }));
      }
      node.earlyFinish = addDays(node.earlyStart, node.effectiveDuration);
    }
  }

  // FS: successorES = predecessorEF + lag
  // SS: successorES = predecessorES + lag
  // FF: successorEF = predecessorEF + lag → successorES = successorEF - duration
  // SF: successorEF = predecessorES + lag → successorES = successorEF - duration
  calculateSuccessorConstraint(pred: CpmNode, edge: CpmEdge): Date {
    switch (edge.type) {
      case 'FS': return addDays(pred.earlyFinish, edge.lagDays);
      case 'SS': return addDays(pred.earlyStart, edge.lagDays);
      case 'FF': return subDays(addDays(pred.earlyFinish, edge.lagDays), successor.effectiveDuration);
      case 'SF': return subDays(addDays(pred.earlyStart, edge.lagDays), successor.effectiveDuration);
    }
  }

  // 3단계: Backward Pass
  backwardPass(nodes: CpmNode[], edges: CpmEdge[], projectEnd: Date): void {
    const reverseSorted = [...topologicalSort(nodes, edges)].reverse();
    for (const node of reverseSorted) {
      const successors = getSuccessors(node.taskId, edges);
      if (successors.length === 0) {
        node.lateFinish = projectEnd;
      } else {
        node.lateFinish = min(successors.map(s => {
          return calculatePredecessorConstraint(node, s, edges);
        }));
      }
      node.lateStart = subDays(node.lateFinish, node.effectiveDuration);
    }
  }

  // 4단계: Float 계산 및 크리티컬 패스 식별
  calculateFloat(nodes: CpmNode[]): void {
    for (const node of nodes) {
      node.totalFloat = dateDiffDays(node.lateStart, node.earlyStart);
      node.isCritical = node.totalFloat === 0;
    }
  }
}
```

### 4.3 영향 분석 알고리즘

```typescript
// domain/algorithms/impact.algorithm.ts

function propagateDelay(
  taskId: string,
  delayDays: number,
  allTasks: Map<string, CpmNode>,
  edges: CpmEdge[]
): ImpactAnalysisResult {
  const affected: AffectedTask[] = [];
  const visited = new Set<string>();
  const queue: { taskId: string; delay: number; chain: string[] }[] = [
    { taskId, delay: delayDays, chain: [taskId] }
  ];

  while (queue.length > 0) {
    const { taskId: currentId, delay, chain } = queue.shift()!;
    const successors = getSuccessors(currentId, edges);

    for (const edge of successors) {
      if (visited.has(edge.successorId)) continue;

      const successor = allTasks.get(edge.successorId)!;
      const propagated = calculatePropagatedDelay(edge.type, delay, edge.lagDays);

      // Float 초과 시에만 실제 지연 발생
      if (propagated > successor.totalFloat) {
        const actualDelay = propagated - successor.totalFloat;
        visited.add(edge.successorId);
        affected.push({
          taskId: edge.successorId,
          taskName: successor.name,
          originalEndDate: successor.earlyFinish,
          projectedEndDate: addDays(successor.earlyFinish, actualDelay),
          propagatedDelayDays: actualDelay,
          isCritical: successor.isCritical,
          dependencyChain: [...chain, edge.successorId],
        });
        queue.push({
          taskId: edge.successorId,
          delay: actualDelay,
          chain: [...chain, edge.successorId],
        });
      }
    }
  }

  // 프로젝트 완료일 재계산
  const newProjectEnd = affected.reduce((maxDate, t) => {
    return t.projectedEndDate > maxDate ? t.projectedEndDate : maxDate;
  }, originalProjectEnd);

  return { affectedTasks: affected, projectEndDateChange: { ... } };
}
```

```typescript
class RedisCache {
  async getOrSet<T>(
    key: string,
    ttl: number,
    factory: () => Promise<T>
  ): Promise<T> {
    const cached = await this.redis.get(key);
    if (cached) return JSON.parse(cached);
    const value = await factory();
    await this.redis.setex(key, ttl, JSON.stringify(value));
    return value;
  }

  async invalidate(pattern: string): Promise<void> {
    const keys = await this.redis.keys(pattern);
    if (keys.length > 0) await this.redis.del(...keys);
  }
}
```

## 8. 그룹 롤업 집계 쿼리

```sql
-- L1 그룹 롤업 (직접 소속 + L2 하위 그룹 소속 프로젝트, DISTINCT)
WITH group_projects AS (
  -- L1 직접 소속
  SELECT m.project_id FROM project_group_memberships m WHERE m.group_id = :l1GroupId
  UNION
  -- L2 경유 소속
  SELECT m.project_id
  FROM project_group_memberships m
  JOIN project_groups g ON g.id = m.group_id
  WHERE g.parent_group_id = :l1GroupId
)
SELECT
  COUNT(DISTINCT p.id)                                        AS total_projects,
  SUM(p.planned_budget)                                       AS planned_budget,
  SUM(p.actual_budget)                                        AS actual_budget,
  SUM(p.planned_budget * p.overall_progress) / NULLIF(SUM(p.planned_budget), 0)
                                                              AS weighted_progress,
  MIN(p.effective_start_date)                                 AS earliest_start,
  MAX(p.effective_end_date)                                   AS latest_end
FROM projects p
WHERE p.id IN (SELECT project_id FROM group_projects);
```
