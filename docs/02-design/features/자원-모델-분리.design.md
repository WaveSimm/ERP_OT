## 1. 설계 요약

`project.resources` 단일 테이블을 3개 모델로 분리하고, 사람 자원은 `auth_users`로 통일.
모든 SegmentAssignment는 polymorphic FK(personUserId / externalPersonId / equipmentResourceId 셋 중 하나)로 변환.

**4 Phase 무손실 마이그레이션** (Phase 1 스키마 확장 → Phase 2 데이터 이전 → Phase 3 코드 전환 → Phase 4 정리).

## 3. 데이터베이스 스키마

### 3.1 services/auth — User 모델 확장

```prisma
// services/auth/prisma/schema.prisma

model User {
  id              String              @id @default(cuid())
  email           String              @unique
  passwordHash    String
  name            String
  role            UserRole            @default(USER)
  // 신규 필드
  status          EmployeeStatus      @default(ACTIVE)
  hireDate        DateTime?           @db.Date
  retirementDate  DateTime?           @db.Date
  loginEnabled    Boolean             @default(true)
  // 기존 관계
  profile         UserProfile?
  refreshTokens   RefreshToken[]
  ...
  @@map("auth_users")
}

enum EmployeeStatus {
  ACTIVE       // 현직 — 모든 기능 사용 가능
  RETIRED      // 퇴직자 — 데이터 보존, 신규 배정·로그인·결재 차단
  SUSPENDED    // 휴직·정지 — 일시적 차단, 복귀 가능
}
```

**status 의미**:
- `ACTIVE` (default): 모든 기능 정상
- `RETIRED`: 새 task 배정 후보·결재라인 후보·사용자 검색·알림 수신에서 자동 제외, 로그인 차단 (`loginEnabled=false`도 자동)
- `SUSPENDED`: 일시 차단. ACTIVE로 다시 돌아올 가능성 명시

**자동 처리**:
- `status=RETIRED` 변경 시 `loginEnabled=false`, `retirementDate=오늘` 자동 채움
- `status=ACTIVE`로 복귀 시 `retirementDate=null`, `loginEnabled=true` 자동
- 로그인 시점에 `status=ACTIVE && loginEnabled=true` 검증 (기존 로직에 추가)

### 3.2 services/project — 모델 분리

```prisma
// services/project/prisma/schema.prisma

// ─── EquipmentResource (Resource → 분리·rename) ────────────
model EquipmentResource {
  id           String        @id @default(cuid())
  name         String
  type         EquipmentType @default(EQUIPMENT)
  isActive     Boolean       @default(true)
  // 향후 확장 가능: serialNumber, location, calibrationCycle, lastInspectedAt, ownerDeptId 등
  createdAt    DateTime      @default(now())
  updatedAt    DateTime      @updatedAt

  groupMemberships ResourceGroupMember[]
  assignments      SegmentAssignment[]

  @@map("equipment_resources")
  @@schema("project")
}

enum EquipmentType {
  EQUIPMENT
  VEHICLE
  FACILITY

  @@schema("project")
}

// ─── ExternalPerson (외주) ──────────────────────────────────
model ExternalPerson {
  id            String         @id @default(cuid())
  name          String
  company       String?        // 소속 업체
  contactEmail  String?
  contactPhone  String?
  status        ExternalStatus @default(ACTIVE)
  contractStart DateTime?      @db.Date
  contractEnd   DateTime?      @db.Date
  notes         String?        @db.Text
  createdAt     DateTime       @default(now())
  updatedAt     DateTime       @updatedAt

  groupMemberships ResourceGroupMember[]
  assignments      SegmentAssignment[]

  @@map("external_persons")
  @@schema("project")
}

enum ExternalStatus {
  ACTIVE      // 계약 중
  ARCHIVED    // 종료

  @@schema("project")
}

// ─── SegmentAssignment polymorphic ──────────────────────────
model SegmentAssignment {
  id                    String         @id @default(cuid())
  segmentId             String
  // polymorphic FK — 셋 중 정확히 하나만 NOT NULL (DB CHECK constraint)
  personUserId          String?        // → auth_users.id (cross-service, FK 없음)
  externalPersonId      String?        // → external_persons.id
  equipmentResourceId   String?        // → equipment_resources.id

  allocationMode        AllocationMode @default(PERCENT)
  allocationPercent     Float?
  allocationHoursPerDay Float?
  createdAt             DateTime       @default(now())
  updatedAt             DateTime       @updatedAt

  segment   TaskSegment        @relation(fields: [segmentId], references: [id], onDelete: Cascade)
  external  ExternalPerson?    @relation(fields: [externalPersonId], references: [id], onDelete: Cascade)
  equipment EquipmentResource? @relation(fields: [equipmentResourceId], references: [id], onDelete: Cascade)

  // unique 제약 — 한 segment에 같은 자원 중복 배정 금지 (3개 컬럼 각각)
  @@unique([segmentId, personUserId])
  @@unique([segmentId, externalPersonId])
  @@unique([segmentId, equipmentResourceId])
  @@map("segment_assignments")
  @@schema("project")
}

// ─── ResourceGroup (type 필드 추가, 단일 종류 그룹 강제) ────
model ResourceGroup {
  id          String           @id @default(cuid())
  name        String
  description String?
  type        ResourceGroupType   // 신규 — PERSON / EXTERNAL / EQUIPMENT
  parentId    String?
  sortOrder   Int              @default(0)
  isActive    Boolean          @default(true)
  createdAt   DateTime         @default(now())
  updatedAt   DateTime         @updatedAt

  parent      ResourceGroup?       @relation("ResourceGroupTree", fields: [parentId], references: [id])
  children    ResourceGroup[]      @relation("ResourceGroupTree")
  memberships ResourceGroupMember[]

  @@map("resource_groups")
  @@schema("project")
}

enum ResourceGroupType {
  PERSON      // 직원 그룹 (예: 부서, 팀)
  EXTERNAL    // 외부 자원 그룹 (예: 외주 업체별)
  EQUIPMENT   // 비인력 자원 그룹 (예: 장비 카테고리)

  @@schema("project")
}

// ─── ResourceGroupMember polymorphic + group.type 일치 ──────
model ResourceGroupMember {
  id                    String   @id @default(cuid())
  groupId               String
  // group.type에 맞는 컬럼만 채움 (application 레이어에서 검증)
  personUserId          String?
  externalPersonId      String?
  equipmentResourceId   String?
  sortOrder             Int      @default(0)
  createdAt             DateTime @default(now())

  group     ResourceGroup      @relation(fields: [groupId], references: [id], onDelete: Cascade)
  external  ExternalPerson?    @relation(fields: [externalPersonId], references: [id], onDelete: Cascade)
  equipment EquipmentResource? @relation(fields: [equipmentResourceId], references: [id], onDelete: Cascade)

  @@unique([groupId, personUserId])
  @@unique([groupId, externalPersonId])
  @@unique([groupId, equipmentResourceId])
  @@map("resource_group_members")
  @@schema("project")
}
```

**그룹 type 강제 (v1.1)**:
- `ResourceGroup.type = PERSON` 그룹은 멤버에 `personUserId`만 채울 수 있음 (다른 두 컬럼 NULL 필수)
- `ResourceGroup.type = EXTERNAL` 그룹은 `externalPersonId`만
- `ResourceGroup.type = EQUIPMENT` 그룹은 `equipmentResourceId`만
- DB CHECK constraint:

```sql
ALTER TABLE project.resource_group_members
  ADD CONSTRAINT resource_group_members_type_match_check
  CHECK (
    -- 정확히 한 컬럼만 NOT NULL (XOR)
    (CASE WHEN person_user_id IS NOT NULL THEN 1 ELSE 0 END) +
    (CASE WHEN external_person_id IS NOT NULL THEN 1 ELSE 0 END) +
    (CASE WHEN equipment_resource_id IS NOT NULL THEN 1 ELSE 0 END) = 1
  );
-- group.type 일치 검증은 application service에서 (cross-table CHECK는 PostgreSQL trigger 또는 service 검증)
```

**추가 검증 (application service)**:
```ts
// addGroupMember(groupId, dto)
const group = await prisma.resourceGroup.findUnique({ where: { id: groupId } });
if (group.type === 'PERSON' && !dto.personUserId)
  throw new AppError(400, 'GROUP_TYPE_MISMATCH', 'PERSON 그룹은 personUserId 필수');
if (group.type === 'EXTERNAL' && !dto.externalPersonId)
  throw new AppError(400, 'GROUP_TYPE_MISMATCH', 'EXTERNAL 그룹은 externalPersonId 필수');
if (group.type === 'EQUIPMENT' && !dto.equipmentResourceId)
  throw new AppError(400, 'GROUP_TYPE_MISMATCH', 'EQUIPMENT 그룹은 equipmentResourceId 필수');
```

**XOR 보장**:
Prisma는 multi-column CHECK constraint를 직접 지원 안 함 → **마이그레이션 SQL에서 직접 추가**:

```sql
ALTER TABLE project.segment_assignments
  ADD CONSTRAINT segment_assignments_xor_check
  CHECK (
    (CASE WHEN person_user_id IS NOT NULL THEN 1 ELSE 0 END) +
    (CASE WHEN external_person_id IS NOT NULL THEN 1 ELSE 0 END) +
    (CASE WHEN equipment_resource_id IS NOT NULL THEN 1 ELSE 0 END) = 1
  );

ALTER TABLE project.resource_group_members
  ADD CONSTRAINT resource_group_members_xor_check
  CHECK (
    (CASE WHEN person_user_id IS NOT NULL THEN 1 ELSE 0 END) +
    (CASE WHEN external_person_id IS NOT NULL THEN 1 ELSE 0 END) +
    (CASE WHEN equipment_resource_id IS NOT NULL THEN 1 ELSE 0 END) = 1
  );
```

### 5.2 Phase 2 데이터 이전 의사코드

```ts
// scripts/migrate-resource-model/02-data-migrate.ts
const isDryRun = process.argv.includes('--dry-run');

async function main() {
  // 1) 사전 데이터 수집
  const allAuthUsers = await authPrisma.user.findMany({ select: { id, email, name } });
  const emailToAuthId = new Map(allAuthUsers.map(u => [u.email, u.id]));

  const personResources = await projectPrisma.resource.findMany({ where: { type: 'PERSON' } });
  const equipmentResources = await projectPrisma.resource.findMany({ where: { type: { in: ['EQUIPMENT', 'VEHICLE', 'FACILITY'] } } });

  // 2) 매핑 검증 (dry-run에서 핵심)
  const unmapped: { id, name, userId }[] = [];
  for (const r of personResources) {
    if (!r.userId || !emailToAuthId.has(r.userId)) {
      unmapped.push({ id: r.id, name: r.name, userId: r.userId });
    }
  }
  if (unmapped.length > 0) {
    console.warn(`⚠️ 매핑 실패 PERSON Resource: ${unmapped.length}건`);
    console.table(unmapped);
    if (!isDryRun) {
      throw new Error('매핑 실패 자원이 있어 마이그레이션 중단. 수동 처리 필요.');
    }
  }

  // 3) EquipmentResource 생성 (Phase 1에서 테이블 만든 후)
  const oldResIdToEquipId = new Map<string, string>();
  if (!isDryRun) {
    for (const r of equipmentResources) {
      const created = await projectPrisma.equipmentResource.create({
        data: {
          name: r.name,
          type: r.type as any,  // EQUIPMENT/VEHICLE/FACILITY
          isActive: r.isActive,
        },
      });
      oldResIdToEquipId.set(r.id, created.id);
    }
  }

  // 4) SegmentAssignment 변환
  const assignments = await projectPrisma.segmentAssignment.findMany();
  const stats = { person: 0, equipment: 0, external: 0, failed: 0 };

  for (const a of assignments) {
    const r = personResources.find(x => x.id === a.resourceId)
           ?? equipmentResources.find(x => x.id === a.resourceId);
    if (!r) { stats.failed++; continue; }

    const updateData: any = {};
    if (r.type === 'PERSON') {
      const authId = r.userId ? emailToAuthId.get(r.userId) : null;
      if (!authId) { stats.failed++; continue; }
      updateData.personUserId = authId;
      stats.person++;
    } else {
      const equipId = oldResIdToEquipId.get(r.id);
      if (!equipId) { stats.failed++; continue; }
      updateData.equipmentResourceId = equipId;
      stats.equipment++;
    }

    if (!isDryRun) {
      await projectPrisma.segmentAssignment.update({
        where: { id: a.id },
        data: updateData,
      });
    }
  }

  // 5) ResourceGroupMember 동일 변환
  // (생략, SegmentAssignment와 동일 패턴)

  console.log('=== 결과 ===');
  console.table(stats);
  if (stats.failed > 0) {
    console.error(`❌ 실패 ${stats.failed}건 — 수동 처리 필요`);
    process.exit(1);
  }
}
```
