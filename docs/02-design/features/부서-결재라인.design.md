## 5. attendance-service 연동 수정

### 5.1 결재자 조회 흐름

```
[휴가 신청]
  1. POST /api/v1/leave/requests (attendance-service)
  2. auth-service 내부 API 호출:
     GET /internal/users/{userId}/approver
  3. 결재자 userId로 알림 발송
  4. LeaveRequest.approverId 초기화 (선택 — 1차 결재자 선저장)

[코드]
// leave.service.ts
async createRequest(userId: string, data: LeaveRequestDto) {
  // 결재자 조회 (캐시 우선)
  const approverInfo = await this.authClient.getApprover(userId);
  const approverId = approverInfo?.delegateId ?? approverInfo?.approverId;

  const request = await this.prisma.leaveRequest.create({
    data: { ...data, userId },
  });

  // 알림 (결재자에게)
  if (approverId) {
    await this.notificationService.create({
      userId: approverId,
      type: "LEAVE_PENDING",
      title: "휴가 승인 요청",
      body: `${data.startDate} ~ ${data.endDate} 휴가 신청이 접수되었습니다.`,
      linkUrl: "/me/team",
    });
  }

  return request;
}
```

### 5.3 AuthClient (attendance-service/infrastructure/)

```typescript
// services/attendance/src/infrastructure/auth-client.ts
export class AuthClient {
  constructor(
    private readonly baseUrl: string,
    private readonly token: string,
    private readonly redis: Redis,
  ) {}

  async getApprover(userId: string): Promise<ApproverInfo | null> {
    const cacheKey = `approver:${userId}`;
    const cached = await this.redis.get(cacheKey);
    if (cached) return JSON.parse(cached);

    const res = await fetch(`${this.baseUrl}/internal/users/${userId}/approver`, {
      headers: { "X-Internal-Token": this.token },
    });
    if (!res.ok) return null;
    const data = await res.json();
    await this.redis.set(cacheKey, JSON.stringify(data), "EX", 300); // 5분 캐시
    return data;
  }

  async getSubordinates(approverId: string): Promise<string[]> {
    const cacheKey = `subordinates:${approverId}`;
    const cached = await this.redis.get(cacheKey);
    if (cached) return JSON.parse(cached);

    const res = await fetch(`${this.baseUrl}/internal/approver/${approverId}/subordinates`, {
      headers: { "X-Internal-Token": this.token },
    });
    if (!res.ok) return [];
    const data = await res.json();
    await this.redis.set(cacheKey, JSON.stringify(data), "EX", 300);
    return data;
  }

  async bulkGetUsers(ids: string[]): Promise<Record<string, { name: string; email: string }>> {
    if (ids.length === 0) return {};
    const res = await fetch(
      `${this.baseUrl}/internal/users/bulk?ids=${ids.join(",")}`,
      { headers: { "X-Internal-Token": this.token } },
    );
    if (!res.ok) return {};
    return res.json();
  }
}
```

export async function internalAuth(req: FastifyRequest, reply: FastifyReply) {
  const token = req.headers["x-internal-token"];
  if (token !== process.env.INTERNAL_API_TOKEN) {
    return reply.status(401).send({ code: "UNAUTHORIZED", message: "내부 API 인증 실패" });
  }
}

// index.ts에 internal routes 등록

  prefix: "/internal",

## 10. 마이그레이션 전략

기존 `UserProfile.departmentId` (문자열)가 있는 데이터를 보존:

```sql
-- migration script (prisma migrate 후 수동 실행)
-- 기존 departmentName 문자열로 Department 레코드 자동 생성
INSERT INTO departments (id, name, code, level)
SELECT DISTINCT
  gen_random_uuid(),
  "department_name",
  upper(replace("department_name", ' ', '_')),
  3
FROM auth_user_profiles
WHERE "department_name" IS NOT NULL
ON CONFLICT DO NOTHING;

-- UserProfile.departmentId 업데이트
UPDATE auth_user_profiles up
SET department_id = d.id
FROM departments d
WHERE up.department_name = d.name
  AND up.department_id IS NULL;
```
